CREATE TABLE Club_members( id INTEGER, name VARCHAR(50), PRIMARY KEY (id))                                                                                                                                CREATE TABLE Events( id INTEGER, name CHAR(255), responsible_id INTEGER, PRIMARY KEY (id), FOREIGN KEY (id)  REFERENCES Club_members (id)  ON DELETE CASCADE)                                                                                                 CREATE TABLE Mentors( id INTEGER, mentor_id INTEGER, mentis_id INTEGER, PRIMARY KEY (id), FOREIGN KEY (mentor_id)      REFERENCES Club_members (id)      ON DELETE CASCADE FOREIGN KEY (mentis_id)      REFERENCES Club_members (id)      ON DELETE CASCADE)                                                                                                                 CREATE TABLE Projects( id INTEGER, name CHAR(255), team_lead_id INTEGER, team_id INTEGER PRIMARY KEY (name), FOREIGN KEY (team_lead_id)      REFERENCES Club_members (id) FOREIGN KEY (team_id)      REFERENCES Club_members (id))                                                                                                                  CREATE TABLE Departments( id INTEGER, name CHAR(255), PRIMARY KEY(id))CREATE TABLE Heads ( id INTEGER, member_id INTEGER, department_id INTEGER,  FOREIGN KEY (member_id)      REFERENCES Club_members (id)      ON DELETE CASCADE FOREIGN KEY (department_id)      REFERENCES Departments (id)      ON DELETE CASCADE)
